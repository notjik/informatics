begin
readln(x);
a := 2*x - 91;

if a > b then

else
b := b - a
end;
writeln(a)

b = 3*x - 159

if a > b:
a -= b
else:

using namespace std;
int main()
{

cin >> x;

a -= b;
else
b -= a;
